238  (defn- on-call
239    "Handle WAMP call (RPC) messages"
240    [callbacks sess-id topic call-id & call-params]
241    (if-let [rpc-cb (callbacks topic)]
242      (try
243        (let [cb-params [sess-id topic call-id call-params]
244              cb-params (apply callback-rewrite (callbacks :on-before) cb-params)
245              [sess-id topic call-id call-params] cb-params
246              rpc-result (binding [*call-sess-id* sess-id]  ; bind optional sess-id
247                           (apply rpc-cb call-params))      ; use fn's own arg signature
248              error      (:error  rpc-result)
249              result     (:result rpc-result)]
250          (if (and (nil? error) (nil? result))
251            ; No map with result or error? Assume successful rpc-result as-is
252            (call-success sess-id topic call-id rpc-result (callbacks :on-after-success))
253            (if (nil? error)
254              (call-success sess-id topic call-id result (callbacks :on-after-success))
255              (call-error   sess-id topic call-id error  (callbacks :on-after-error)))))
256  
257        (catch Exception e
258          (call-error sess-id topic call-id
259            {:uri URI-WAMP-ERROR-INTERNAL
260             :message DESC-WAMP-ERROR-INTERNAL
261             :description (.getMessage e)}
262            (callbacks :on-after-error))))
263  
264      (call-error sess-id topic call-id
265        {:uri URI-WAMP-ERROR-NOTFOUND
266         :message DESC-WAMP-ERROR-NOTFOUND}
267        (callbacks :on-after-error))))

269  (defn- map-key-or-prefix
270    "Finds a map value by key or lookup by string key prefix (ending with *)."
271    [m k]
272    (if-let [v (m k)] v
273      (some #(when (not (nil? %)) %)
274        (for [[mk mv] m]
275          (when (and (not (keyword? mk)) (not (false? mv))
276                  (= \* (last mk))
277                  (= (take (dec (count mk)) k) (butlast mk)))
278            mv)))))

280  (defn- on-subscribe
281    [callbacks sess-id topic]
282    (dosync
283      (when (nil? (get-in @topic-clients [topic sess-id]))
284        (when-let [topic-cb (map-key-or-prefix callbacks topic)]
285          (when (or (true? topic-cb) (topic-cb sess-id topic))
286            (let [on-after-cb (callbacks :on-after)]
287              (topic-subscribe topic sess-id)
288              (when (fn? on-after-cb)
289                (on-after-cb sess-id topic))))))))

296  (defn- on-publish
297    "Handles WAMP publish messages, sending event messages back out
298    to clients subscribed to the topic.
299    [ TYPE_ID_PUBLISH , topicURI , event [, exclude [, eligible ]]"
300    ([callbacks sess-id topic event]
301      (on-publish callbacks sess-id topic event false nil))
302    ([callbacks sess-id topic event exclude]
303      (on-publish callbacks sess-id topic event exclude nil))
304    ([callbacks sess-id topic event exclude eligible]
305      (when-let [pub-cb (map-key-or-prefix callbacks topic)]
306        (let [cb-params [sess-id topic event exclude eligible]
307              cb-params (apply callback-rewrite pub-cb cb-params)
308              on-after-cb (callbacks :on-after)]
309          (when (sequential? cb-params)
310            (let [[sess-id topic event exclude eligible] cb-params
311                  exclude (get-publish-exclude sess-id exclude)]
312              (if-not (nil? eligible)
313                (emit-event! topic event eligible)
314                (broadcast-event! topic event exclude))
315              (when (fn? on-after-cb)
316                (on-after-cb sess-id topic event exclude eligible))))))))

318  (defn- on-message
319    "Handles all http-kit messages. parses the incoming data as json
320    and finds the appropriate wamp callback."
321    [sess-id callbacks]
322    (fn [data]
323      (log/trace "Data received:" data)
324      (let [[msg-type & msg-params] (try (json/decode data)
325                                      (catch com.fasterxml.jackson.core.JsonParseException ex
326                                        [nil nil]))
327            on-call-cbs  (callbacks :on-call)
328            on-sub-cbs   (callbacks :on-subscribe)
329            on-unsub-cb  (callbacks :on-unsubscribe)
330            on-pub-cbs   (callbacks :on-publish)]
331        (case msg-type
332  
333          1 ;TYPE-ID-PREFIX
334          (apply add-topic-prefix sess-id msg-params)
335  
336          2 ;TYPE-ID-CALL
337          (if (map? on-call-cbs)
338            (let [[call-id topic-uri & call-params] msg-params
339                  topic (get-topic sess-id topic-uri)]
340              (apply on-call on-call-cbs sess-id topic call-id call-params)))
341  
342          5 ;TYPE-ID-SUBSCRIBE
343          (let [topic (get-topic sess-id (first msg-params))]
344            (on-subscribe on-sub-cbs sess-id topic))
345  
346          6 ;TYPE-ID-UNSUBSCRIBE
347          (let [topic (get-topic sess-id (first msg-params))]
348            (dosync
349              (when (true? (get-in @topic-clients [topic sess-id]))
350                (topic-unsubscribe topic sess-id)
351                (when (fn? on-unsub-cb) (on-unsub-cb sess-id topic)))))
352  
353          7 ;TYPE-ID-PUBLISH
354          (let [[topic-uri event & pub-args] msg-params
355                topic (get-topic sess-id topic-uri)]
356            (apply on-publish on-pub-cbs sess-id topic event pub-args))
357  
358          ; default: Unknown message type
359          (log/warn "Unknown message type" data)))))

362  (defn http-kit-handler
363    "Sets up the necessary http-kit websocket event handlers
364    for use with the WAMP sub-protocol. Returns a WAMP client session id.
365  
366    Example usage:
367  
368      (http-kit/with-channel req channel
369        (if-not (:websocket? req)
370          (http-kit/close channel)
371          (http-kit-handler channel
372            {:on-open        on-open-fn
373             :on-close       on-close-fn
374  
375             :on-call        {(rpc-url \"add\")      +         ; map topics to rpc functions
376                              (rpc-url \"echo\")     identity
377                              :on-before           on-before-call-fn
378                              :on-after-error      on-after-call-error-fn
379                              :on-after-success    on-after-call-success-fn}
380  
381             :on-subscribe   {(evt-url \"chat\")     on-subscribe-fn? ; allowed to subscribe?
382                              (evt-url \"prefix*\")  true             ; match topics by prefix
383                              (evt-url \"sub-only\") true             ; implicitly allowed
384                              (evt-url \"pub-only\") false            ; subscription is denied
385                              :on-after            on-after-subscribe-fn};
386  
387             :on-publish     {(evt-url \"chat\")     on-publish-fn   ; custom event broker
388                              (evt-url \"prefix*\")  true            ; pass events through as-is
389                              (evt-url \"sub-only\") false           ; publishing is denied
390                              (evt-url \"pub-only\") true
391                              :on-after            on-after-publish-fn}
392  
393             :on-unsubscribe on-unsubscribe-fn})))
394  
395    Callback signatures:
396  
397      (on-open-fn sess-id)
398      (on-close-fn sess-id status)
399  
400      (rpc-call ...)
401        Can have any signature. The parameters received from the client will be applied as-is.
402        The client session is also available in the bound *call-sess-id* var.
403        The function may return a value as is, or in a result map: {:result \"my result\"},
404        or as an error map: {:error {:uri \"http://example.com/error#give-error\"
405                                     :message \"Test error\"
406                                     :description \"Test error description\"}}
407  
408      (on-before-call-fn sess-id topic call-id call-params)
409        To allow call, return params as vector: [sess-id topic call-id call-params]
410        To deny, return nil/false.
411  
412      (on-after-call-error-fn sess-id topic call-id error)
413        Return params as vector: [sess-id topic call-id error]
414  
415      (on-after-call-success-fn sess-id topic call-id result)
416        Return params as vector: [sess-id topic call-id result]
417  
418      (on-subscribe-fn? sess-id topic)
419        Return true to allow client to subscribe, false to deny.
420
